- If you are currently using conda on your machine run

conda create -n cs168 python=3.6
source activate cs168
# cd into the directory containing the hw1 files then run
pip install -r requirements.txt
# once installed run
jupyter notebook

- If you are using virtualenv and python on your machine
# cd into directory containing hw1 files..
virtualenv cs168
# cd into cs168 folder
cd ./cs168/bin
source activate
pip install -r requirements.txt
jupyter notebook
